Find Relation

        const findRelation = (x,y) => {
            if (x < y) {
                return x+" is smaller than "+y;
            }
            else if (x > y) {
                return x+" is greater than "+y;
            }
            else {
                return x+" is equal to "+y;
            }
        };

        const Max_out_of_three = (A,B,C) => {
            if (A===B && A===C) {
                return -1;
            }
            else if (A>B && A>C) {
                return A;
            }

                return C;

Second Smallest

        const findSndSmallest = (x,y,z) => {
            if (x != y && y!=z && x!=z) {
                if ((x < y && x > z) || (x > y && x < z)) {
                    return x;
                }
                else if ((y < x && y > z) || (y > x && y < z)) {
                    return y;
                }
                else {
                    return z;
                }
            }  
        };
